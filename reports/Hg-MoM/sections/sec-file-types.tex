\section{File Types}
Standard file types
\begin{enumerate}
	\item \texttt{*.obj}
	\item \texttt{*.txt}
\end{enumerate}

Intrinsic file types
\begin{enumerate}
	\item \texttt{*.geo}
	\item \texttt{*.facet}
\end{enumerate}
