% Beamer Presentation Mode and Themes
\mode<presentation> {
    \usetheme{Copenhagen}                 % Main theme
    \usecolortheme{beaver}                % Color theme
    \usefonttheme[onlysmall]{structurebold} % Bold structure for small text
    \usefonttheme[onlymath]{serif}        % Serif for math fonts
    % \usepackage{beamerthemesplit}         % Custom appearance (activate if needed)
}
\renewcommand{\seriesdefault}{\bfdefault} % Default font series

% Ensure normal text color is black
\setbeamercolor{normal text}{fg=black}

% Enable Slide Numbers
\input{\pGlobalSetup slide-numbers}

% Hide navigation symbols
% \setbeamertemplate{navigation symbols}{}

% Define Custom Colors
\definecolor{airforceblue}{RGB}{0, 48, 143} % Custom Air Force blue

% Set Beamer Colors
\setbeamercolor{normal text}{fg=black}             % Body text
\setbeamercolor{structure}{fg=airforceblue}        % Structure (e.g., section/subsection)
\setbeamercolor{frametitle}{fg=black}              % Slide titles
\setbeamercolor{title}{fg=black}                   % Main title
\setbeamercolor{section in head/foot}{fg=airforceblue} % Section in navigation panel
\setbeamercolor{subsection in head/foot}{fg=airforceblue} % Subsection in navigation panel
\setbeamercolor{navigation symbols}{fg=airforceblue} % Optional: Navigation symbols
