% Maxwell equations
% 	Components
\newcommand{\divE}[0]						{\href{https://en.wikipedia.org/wiki/Gauss\%27s\_law}{\nabla \cdot \E{}}}
\newcommand{\divB}[0]						{\href{https://en.wikipedia.org/wiki/Gauss\%27s\_law}{\nabla \cdot \B{}}}

\newcommand{\divErhs}[0]					{4\pi \rho}
\newcommand{\divBrhs}[0]					{0}

\newcommand{\curlE}[0]						{\href{https://en.wikipedia.org/wiki/Faraday\%27s\_law\_of\_induction}{\nabla \times \E{}}}
\newcommand{\curlB}[0]						{\href{https://en.wikipedia.org/wiki/Amp\%C3\%A8re\%27s\_circuital\_law}{\nabla \times \B{}}}

\newcommand{\curlElhs}[0]					{\curlE + \tfrac{1}{c}\partial_{t} \B{}}
\newcommand{\curlBlhs}[0]					{\curlB - \tfrac{1}{c}\partial_{t} \E{}}

\newcommand{\curlErhs}[0]					{0}
\newcommand{\curlBrhs}[0]					{\frac{4 \pi}{c}\jay{}}

\newcommand{\maxwellFaraday}[0]				{\divE = -\partial_{t} \B{}}

% bold letter versions
\newcommand{\divEb}[0]						{\nabla \cdot \mathbf{E}}
\newcommand{\divBb}[0]						{\nabla \cdot \mathbf{B}}

\newcommand{\curlEb}[0]						{\nabla \times \mathbf{E}}
\newcommand{\curlBb}[0]						{\nabla \times \mathbf{B}}

\newcommand{\curlEblhs}[0]					{\curlEb + \tfrac{1}{c}\partial_{t} \mathbf{B}}
\newcommand{\curlBblhs}[0]					{\curlBb - \tfrac{1}{c}\partial_{t} \mathbf{E}}

% 	Equations
\newcommand{\maxwellEquationi}[0]				{\divE = \divErhs}
\newcommand{\maxwellEquationii}[0]			{\curlElhs = \curlErhs}
\newcommand{\maxwellEquationiii}[0]			{\divB = \divBrhs}
\newcommand{\maxwellEquationiv}[0]			{\curlBlhs = \curlBrhs}

\newcommand{\maxwellFaradayb}[0]			{\divEb = -\partial_{t} \mathbf{B}}

\newcommand{\maxwellDifferentialCGS }[0]		{\begin{array}{rcl c rcl}
		%
	\divE &=& \divErhs &\qquad &\curlElhs &=& \curlErhs \\[15pt]
	\divB &=& \divBrhs &		    &\curlBlhs &=& \curlBrhs
			%
\end{array}
}
