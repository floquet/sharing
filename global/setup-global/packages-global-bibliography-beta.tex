% Load the biblatex package with desired options
% Options:
%   - style=numeric: Produces numeric citations like [1].
%   - backend=biber: Uses Biber for advanced bibliography processing.
%   - sorting=nyt: Sorts entries by Name, Year, and Title.
%   - maxnames=3: Limits the number of authors displayed; "et al." is used for longer lists.
%   - minnames=1: Ensures at least one author is displayed in shortened lists.

% Load the biblatex package with desired options
\usepackage[
    style=numeric,          
    backend=biber,          
    sorting=nyt,            
    maxnames=3,             
    minnames=1              
]{biblatex}

% Configure bibliography layout and formatting
\setlength{\bibhang}{2em}       % Hanging indent of 2em for each entry
\setlength{\bibitemsep}{0.5em}  % Space of 0.5em between entries
\setlength{\bibparsep}{0.2em}   % Space of 0.2em between paragraphs in an entry

% Custom bibliography environment
\defbibenvironment{bibliography}
  {\list{}
    {\setlength{\leftmargin}{\bibhang}%
     \setlength{\itemindent}{-\leftmargin}%
     \setlength{\itemsep}{\bibitemsep}%
     \setlength{\parsep}{\bibparsep}}%
  }
  {\endlist}
  {\item}
