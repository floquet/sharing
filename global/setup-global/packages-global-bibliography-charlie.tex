% Load BibLaTeX with desired options
\usepackage[
    style=numeric,          % Citation style (e.g., numeric, authoryear, etc.)
    backend=biber,          % Use biber for processing
    %sorting=none,            % Use 'none' to follow citation order
    sorting=nyt,            % Sort entries by Name, Year, and Title
    maxnames=3,             % Maximum number of author names to display
    minnames=1              % Minimum number of author names to display
]{biblatex}

% Suppress specific fields globally
\AtEveryBibitem{%
    \clearfield{eprint}    % Suppress eprint field
    \clearfield{url}       % Suppress url field
}

% Configure bibliography layout and formatting
\setlength{\bibhang}{2em}       % Hanging indent of 2em for each entry
\setlength{\bibitemsep}{0.5em}  % Space of 0.5em between entries
\setlength{\bibparsep}{0.2em}   % Space of 0.2em between paragraphs in an entry

% Custom environment for bibliography layout
\defbibenvironment{bibliography}
  {\list{}%
    {%
     \setlength{\leftmargin}{\bibhang}% Hanging indent
     \setlength{\itemindent}{0pt}% No additional indentation for numbers
     \setlength{\itemsep}{\bibitemsep}% Space between items
     \setlength{\parsep}{\bibparsep}% Space between paragraphs in entries
    }%
  }%
  {\endlist}
  {\item[\printfield{labelnumber}]} % Print the numeric label for each entry

\endinput  % == End of file ==
