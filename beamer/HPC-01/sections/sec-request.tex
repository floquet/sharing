    \centering
\begin{enumerate}
	\item Contract number
	\item Contract expiration
	\item Government sponsor 
	\begin{enumerate}
		\item Name
		\item eMail
		\item Phone number
	\end{enumerate}
	\item Business address
	\item FSO eMail
	\item Your eMail

	\item \mg{\href{https://militarycac.com/}{Middleware}: CAC reader}

		\item How to use modules
		\item How to batch
		\item Simple scripts (.bashrc)
		\item Compilers and software
		\item Parallel programming
	\end{enumerate}
	\end{enumerate}
\end{frame}
