\begin{theorem}[\href{https://en.wikipedia.org/wiki/Riesz\%E2\%80\%93Fischer_theorem}{Riesz-Fischer}]

Let $\lst{\phi_{n}}$ be an orthonormal sequence of functions on $\Omega$ and suppose $\sum \abs{a_{n}}^{2}$ converges. Denote the partial sum as
%
\begin{equation*}
	s_{\tau} = a_{0}\phi_{0} + a_{1}\phi_{1} + \cdots + a_{\tau}\phi_{\tau} .
%\label{eq:}
\end{equation*}
%
There \bl{exists} a function $\bl{F\in\spaceL}$ such that $\lst{s_{\tau}}$ converges to $F$ in $\spaceL$, and such that
%
\begin{equation*}
	F = \sum_{k=0}^{\infty} a_{k}\phi_{k},
%\label{eq:}
\end{equation*}
%
almost everywhere.

\label{thm:RieszFischer}
\end{theorem}
