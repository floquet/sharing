\documentclass{standalone}
\usepackage{tikz}
\begin{document}

\begin{tikzpicture}[ scale = 1 ] % Scale the image to about 5 inches

    % Draw the solid circle outline centered at (0,0)
    \draw[thick] (0,0) circle[radius=2];

    % Dashed part of the ellipse (inside the circle)
    \begin{scope}
        \clip (0,0) circle(2); % Clip to the inside of the circle
        \draw[thick, dashed] (-1.5, 0) ellipse (3cm and 1.5cm);
    \end{scope}

    % Solid part of the ellipse (outside the circle only)
    \begin{scope}
        \clip (-4, -4) rectangle (4, 4); % Overall drawing region
        \clip (-1.5, 0) ellipse (3cm and 1.5cm); % Clip to the outside of the circle
        \draw[thick] (-1.5, 0) ellipse (3cm and 1.5cm);
    \end{scope}

    % Draw vector from the circle's origin to the upper intersection point at 135 degrees
    \draw[thick, -{stealth}] (0,0) -- ++(135:2) node[above right] {$r_0$};

    % Draw the angle arc for f (from 0 to r_0)
    \draw[thick] (0.6, 0) arc[start angle=0, end angle=135, radius=0.6];
    \node at (1, 0.5) {$f$};

    % Draw the angle arc for φ (from r_0 to π)
    \draw[thick] (135:0.8) arc[start angle=135, end angle=180, radius=0.8];
    \node at (-1.5, 0.5) {$\phi$};

    % Dotted horizontal line from left edge of ellipse to right edge of circle
    \draw[dotted] (-3,0) -- (2,0);

\end{tikzpicture}

\end{document}
